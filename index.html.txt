<script src="https://cdnjs.cloudflare.com/ajax/libs/three.js/r128/three.min.js"></script>

<script>
    let scene, camera, renderer, playerCar, road;
    let gameActive = false;
    let score = 0;
    let enemies = [];
    let speed = 0.8;
    let moveL = false, moveR = false;

    function init() {
        scene = new THREE.Scene();
        scene.fog = new THREE.Fog(0x000000, 10, 50);

        camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);
        camera.position.set(0, 3, 8);
        camera.lookAt(0, 0, -5);

        renderer = new THREE.WebGLRenderer({ antialias: true });
        renderer.setSize(window.innerWidth, window.innerHeight);
        document.body.appendChild(renderer.domElement);

        // Lights
        const ambient = new THREE.AmbientLight(0x404040, 2);
        scene.add(ambient);
        const sun = new THREE.PointLight(0x00f3ff, 1, 100);
        sun.position.set(0, 10, 0);
        scene.add(sun);

        // Road
        const roadGeo = new THREE.PlaneGeometry(10, 2000);
        const roadMat = new THREE.MeshStandardMaterial({ color: 0x111111 });
        road = new THREE.Mesh(roadGeo, roadMat);
        road.rotation.x = -Math.PI / 2;
        scene.add(road);

        // Grid lines for speed effect
        const grid = new THREE.GridHelper(200, 40, 0x00f3ff, 0x222222);
        grid.position.y = 0.05;
        scene.add(grid);

        // Player Car
        const carGeo = new THREE.BoxGeometry(1.2, 0.6, 2);
        const carMat = new THREE.MeshStandardMaterial({ color: 0x00f3ff, emissive: 0x00f3ff, emissiveIntensity: 0.5 });
        playerCar = new THREE.Mesh(carGeo, carMat);
        playerCar.position.y = 0.6;
        scene.add(playerCar);

        animate();
    }

    function spawnEnemy() {
        const geo = new THREE.BoxGeometry(1.2, 0.6, 2);
        const mat = new THREE.MeshStandardMaterial({ color: 0xff00ff, emissive: 0xff00ff });
        const enemy = new THREE.Mesh(geo, mat);
        enemy.position.set((Math.random() - 0.5) * 8, 0.6, -100);
        scene.add(enemy);
        enemies.push(enemy);
    }

    function startGame() {
        document.getElementById('menu').style.display = 'none';
        gameActive = true;
        score = 0;
        speed = 0.8;
        playerCar.position.x = 0;
    }

    function animate() {
        requestAnimationFrame(animate);
        
        if (gameActive) {
            // Movement
            if (moveL && playerCar.position.x > -4) playerCar.position.x -= 0.15;
            if (moveR && playerCar.position.x < 4) playerCar.position.x += 0.15;

            // Speed effect
            road.position.z = (road.position.z + speed) % 10;

            // Enemies
            for (let i = enemies.length - 1; i >= 0; i--) {
                enemies[i].position.z += speed;

                // Collision Detection
                const dist = playerCar.position.distanceTo(enemies[i].position);
                if (dist < 1.5) {
                    gameActive = false;
                    document.getElementById('menu').style.display = 'flex';
                    document.querySelector('h1').innerText = "WASTED";
                }

                if (enemies[i].position.z > 10) {
                    scene.remove(enemies[i]);
                    enemies.splice(i, 1);
                    score++;
                    document.getElementById('score').innerText = score;
                    speed += 0.005;
                }
            }

            if (Math.random() < 0.03) spawnEnemy();
        }

        renderer.render(scene, camera);
    }

    // Controls
    window.onkeydown = (e) => { if(e.key==="ArrowLeft") moveL=true; if(e.key==="ArrowRight") moveR=true; };
    window.onkeyup = (e) => { if(e.key==="ArrowLeft") moveL=false; if(e.key==="ArrowRight") moveR=false; };
    
    document.getElementById('lBtn').ontouchstart = () => moveL = true;
    document.getElementById('lBtn').ontouchend = () => moveL = false;
    document.getElementById('rBtn').ontouchstart = () => moveR = true;
    document.getElementById('rBtn').ontouchend = () => moveR = false;

    window.addEventListener('resize', () => {
        camera.aspect = window.innerWidth / window.innerHeight;
        camera.updateProjectionMatrix();
        renderer.setSize(window.innerWidth, window.innerHeight);
    });

    init();
</script>
