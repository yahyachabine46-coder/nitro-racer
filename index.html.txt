        #wrapper { position: absolute; top: 50%; left: 50%; transform: translate(-50%, -50%); width: 95vw; aspect-ratio: 16/9; background: #050505; border: 2px solid #333; }
        canvas { width: 100%; height: 100%; display: block; }
        .ui { position: absolute; inset: 0; display: flex; flex-direction: column; align-items: center; justify-content: center; background: rgba(0,0,0,0.85); z-index: 100; }

<div id="wrapper">
    <div class="stats" id="s">LEVEL 1 | $0 | LIVES: 1</div>
    <div id="m" class="ui">
        <h1 id="t" style="font-size: 50px; color: #ff00ff;">NITRO RACER</h1>
        <button onclick="boot()">START RACE</button>
        <p style="margin-top:20px; color:#888;">ARROWS to steer | SHIFT for Nitro</p>
    </div>
    <canvas id="c"></canvas>

<script>
    const cvs = document.getElementById('c');
    const ctx = cvs.getContext('2d');
    const statBox = document.getElementById('s');
    const menu = document.getElementById('m');
    const title = document.getElementById('t');

    cvs.width = 1600;
    cvs.height = 900;

    let active = false, lane = 2, playerX = 800, playerY = 750, speed = 0;
    let enemies = [], ramps = [], frames = 0, lives = 1, cash = 0;
    let curve = 0, curveTarget = 0, isJumping = 0; // 0 = on ground, >0 = in air
    const LANES = [300, 550, 800, 1050, 1300];

    function drawBox(x, y, w, h, color, curveOff = 0) {
        ctx.fillStyle = color;
        ctx.fillRect(x - w/2 + curveOff, y - h/2, w, h);
    }

    function boot() {
        active = true;
        speed = 12;
        lane = 2;
        playerX = 800;
        enemies = [];
        ramps = [];
        lives = 1;
        frames = 0;
        menu.style.display = 'none';
        run();
    }

    function run() {
        if (!active) return;
        frames++;

        // 1. Road Logic
        if (frames % 120 === 0) curveTarget = (Math.random() - 0.5) * 60;
        curve += (curveTarget - curve) * 0.05;

        // 2. Background & Road
        ctx.fillStyle = "#050505";
        ctx.fillRect(0, 0, 1600, 900);
        
        // Perspective Road
        ctx.fillStyle = "#111";
        ctx.beginPath();
        ctx.moveTo(500 + curve, 0); ctx.lineTo(1100 + curve, 0);
        ctx.lineTo(1500, 900); ctx.lineTo(100, 900);
        ctx.fill();

        // 3. Player Movement
        let targetX = LANES[lane];
        playerX += (targetX - playerX) * 0.15;
        playerX -= curve * 0.1; // Pulling force of the curve
        speed += 0.002;

        // Jump Physics
        if (isJumping > 0) isJumping--;

        // 4. Spawning
        if (Math.random() < 0.03) enemies.push({x: LANES[Math.floor(Math.random()*5)], y: -100});
        if (Math.random() < 0.005) ramps.push({x: LANES[Math.floor(Math.random()*5)], y: -100});

        // 5. Update Ramps
        for (let i = ramps.length - 1; i >= 0; i--) {
            let r = ramps[i];
            r.y += speed;
            let off = (1 - (r.y / 900)) * curve;
            drawBox(r.x + off, r.y, 120, 60, "#ffcc00"); // Yellow Ramp
            
            if (Math.abs(r.x + off - playerX) < 80 && Math.abs(r.y - playerY) < 50 && isJumping === 0) {
                isJumping = 60; // Jump for 60 frames
                ramps.splice(i, 1);
            }
            if (r.y > 1000) ramps.splice(i, 1);
        }

        // 6. Update Enemies
        for (let i = enemies.length - 1; i >= 0; i--) {
            let e = enemies[i];
            e.y += speed;
            let off = (1 - (e.y / 900)) * curve;
            drawBox(e.x + off, e.y, 90, 160, "#f30");

            // Collision (Only if NOT jumping)
            if (isJumping === 0) {
                if (Math.abs(e.x + off - playerX) < 80 && Math.abs(e.y - playerY) < 120) {
                    lives--;
                    enemies.splice(i, 1);
                    if (lives <= 0) die();
                }
            }
            if (e.y > 1000) { enemies.splice(i, 1); cash += 10; }
        }

        // 7. Draw Player
        let pSize = isJumping > 0 ? 1.4 : 1.0; // Scale up when jumping
        ctx.save();
        ctx.translate(playerX, playerY);
        ctx.scale(pSize, pSize);
        drawBox(0, 0, 100, 180, isJumping > 0 ? "#fff" : "#00f2ff");
        ctx.restore();

        statBox.innerText = `LEVEL ${Math.floor(frames/1000)+1} | $${cash} | LIVES: ${lives} ${isJumping > 0 ? " - AIRBORNE!" : ""}`;

        requestAnimationFrame(run);
    }

    function die() {
        active = false;
        menu.style.display = 'flex';
        title.innerText = "CRASHED!";
    }

    window.addEventListener('keydown', e => {
        if ((e.key === "ArrowLeft" || e.key === "a") && lane > 0) lane--;
        if ((e.key === "ArrowRight" || e.key === "d") && lane < 4) lane++;
    });
</script>
