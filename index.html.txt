<script src="https://cdnjs.cloudflare.com/ajax/libs/three.js/r128/three.min.js"></script>

<script>
    let scene, camera, renderer, car, road;
    let isPlaying = false;
    let speed = 0;
    let moveDir = 0; // -1 for left, 1 for right

    function init() {
        // 1. SCENE & CAMERA (3D Setup)
        scene = new THREE.Scene();
        scene.background = new THREE.Color(0x87ceeb); // Sky blue
        
        camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);
        camera.position.set(0, 4, 10); // Placed behind the car
        camera.lookAt(0, 2, -5);

        renderer = new THREE.WebGLRenderer({ antialias: true });
        renderer.setSize(window.innerWidth, window.innerHeight);
        document.body.appendChild(renderer.domElement);

        // 2. LIGHTING
        const light = new THREE.DirectionalLight(0xffffff, 1);
        light.position.set(5, 10, 7);
        scene.add(light);
        scene.add(new THREE.AmbientLight(0xffffff, 0.5));

        // 3. THE ROAD
        const roadGeo = new THREE.PlaneGeometry(12, 1000);
        const roadMat = new THREE.MeshStandardMaterial({ color: 0x444444 });
        road = new THREE.Mesh(roadGeo, roadMat);
        road.rotation.x = -Math.PI / 2;
        scene.add(road);

        // 4. THE CAR
        const carGroup = new THREE.Group();
        const bodyGeo = new THREE.BoxGeometry(1.8, 0.8, 3.5);
        const bodyMat = new THREE.MeshStandardMaterial({ color: 0xffffff });
        const body = new THREE.Mesh(bodyGeo, bodyMat);
        carGroup.add(body);
        
        car = carGroup;
        car.position.y = 0.5;
        scene.add(car);

        // 5. ENVIRONMENT (Palm Trees / Poles)
        for(let i=0; i<40; i++) {
            const pole = new THREE.Mesh(new THREE.CylinderGeometry(0.2, 0.2, 8), new THREE.MeshStandardMaterial({color: 0x654321}));
            pole.position.set(i%2==0 ? 8 : -8, 4, -i * 25);
            scene.add(pole);
        }

        loop();
    }

    function play() {
        document.getElementById('pause-screen').style.display = 'none';
        isPlaying = true;
    }

    function loop() {
        requestAnimationFrame(loop);
        if(isPlaying) {
            speed = Math.min(speed + 0.2, 160); // Max speed 160
            document.getElementById('kmh').innerText = Math.floor(speed);

            // Side movement
            car.position.x += moveDir * 0.15;
            if(car.position.x > 4.5) car.position.x = 4.5;
            if(car.position.x < -4.5) car.position.x = -4.5;

            // Endless Road logic
            road.position.z += speed / 40;
            if(road.position.z > 200) road.position.z = 0;
            
            // Camera subtle shake
            camera.position.y = 4 + Math.sin(Date.now()*0.01)*0.05;
        }
        renderer.render(scene, camera);
    }

    // Input
    window.onkeydown = (e) => {
        if(e.key === "ArrowLeft") moveDir = -1;
        if(e.key === "ArrowRight") moveDir = 1;
    };
    window.onkeyup = () => moveDir = 0;

    init();
</script>
