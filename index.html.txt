<script src="https://cdnjs.cloudflare.com/ajax/libs/three.js/r128/three.min.js"></script>

<script>
    let scene, camera, renderer, car, road;
    let gameActive = false;
    let speed = 0;
    let moveLeft = false, moveRight = false;
    let enemies = [];

    function init() {
        scene = new THREE.Scene();
        scene.background = new THREE.Color(0x87ceeb); // Light blue sky like your image
        
        camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);
        camera.position.set(0, 4, 10); // Third person view
        camera.lookAt(0, 2, -5);

        renderer = new THREE.WebGLRenderer({ antialias: true });
        renderer.setSize(window.innerWidth, window.innerHeight);
        document.body.appendChild(renderer.domElement);

        // Lights
        const hemiLight = new THREE.HemisphereLight(0xffffff, 0x444444, 1);
        scene.add(hemiLight);

        // Road with simple textures
        const roadGeo = new THREE.BoxGeometry(10, 0.5, 1000);
        const roadMat = new THREE.MeshStandardMaterial({ color: 0x333333 });
        road = new THREE.Mesh(roadGeo, roadMat);
        scene.add(road);

        // The Car (Sleek Model)
        const carBody = new THREE.BoxGeometry(1.5, 0.7, 3);
        const carMat = new THREE.MeshStandardMaterial({ color: 0xffffff });
        car = new THREE.Mesh(carBody, carMat);
        car.position.y = 0.6;
        scene.add(car);

        // Visual environment (Trees/Poles)
        for(let i=0; i<50; i++) {
            const poleGeo = new THREE.CylinderGeometry(0.1, 0.1, 10);
            const pole = new THREE.Mesh(poleGeo, new THREE.MeshStandardMaterial({color: 0x555555}));
            pole.position.set(i%2==0 ? 6 : -6, 5, -i*20);
            scene.add(pole);
        }

        animate();
    }

    function resumeGame() {
        document.getElementById('menu').style.display = 'none';
        gameActive = true;
    }

    function animate() {
        requestAnimationFrame(animate);

        if(gameActive) {
            speed = Math.min(speed + 0.2, 180); // Accel to 180km/h
            document.getElementById('speedText').innerText = Math.floor(speed);

            // Car logic
            if(moveLeft && car.position.x > -4) car.position.x -= 0.15;
            if(moveRight && car.position.x < 4) car.position.x += 0.15;

            // Move the world
            road.position.z += speed / 50;
            if(road.position.z > 400) road.position.z = 0;
        }

        renderer.render(scene, camera);
    }

    // Input
    window.onkeydown = (e) => {
        if(e.key === "ArrowLeft") moveLeft = true;
        if(e.key === "ArrowRight") moveRight = true;
    };
    window.onkeyup = (e) => {
        if(e.key === "ArrowLeft") moveLeft = false;
        if(e.key === "ArrowRight") moveRight = false;
    };

    window.onresize = () => {
        camera.aspect = window.innerWidth / window.innerHeight;
        camera.updateProjectionMatrix();
        renderer.setSize(window.innerWidth, window.innerHeight);
    };

    init();
</script>
