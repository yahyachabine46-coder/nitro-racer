    <canvas id="gameCanvas" width="350" height="600"></canvas>
    
    <div id="overlay">
        <h1>Nitro Racer</h1>
        <div class="line"></div>
        <button id="startBtn">Start Engine</button>
        <div class="controls-hint">ARROWS or TOUCH to drive</div>
    </div>

<script>
    const canvas = document.getElementById('gameCanvas');
    const ctx = canvas.getContext('2d');
    const overlay = document.getElementById('overlay');
    const startBtn = document.getElementById('startBtn');

    // Game Constants
    const LANES = [50, 150, 250];
    const CAR_WIDTH = 50;
    const CAR_HEIGHT = 90;

    // Game State
    let gameActive = false;
    let score = 0;
    let bestScore = localStorage.getItem('nitroBest') || 0;
    let player = { lane: 1, x: LANES[1], y: 480 };
    let enemies = [];
    let speed = 5;
    let frame = 0;

    function init() {
        score = 0;
        speed = 5;
        enemies = [];
        player.lane = 1;
        player.x = LANES[player.lane];
        gameActive = true;
        overlay.style.display = 'none';
        animate();
    }

    function drawCar(x, y, color, isPlayer = false) {
        ctx.save();
        ctx.shadowBlur = 15;
        ctx.shadowColor = color;
        
        // Body
        ctx.fillStyle = color;
        ctx.beginPath();
        ctx.roundRect(x - CAR_WIDTH/2, y, CAR_WIDTH, CAR_HEIGHT, 8);
        ctx.fill();

        // Windshield
        ctx.fillStyle = '#111';
        ctx.fillRect(x - 15, y + 15, 30, 20);

        // Tail lights (red)
        ctx.fillStyle = '#ff0000';
        ctx.fillRect(x - 20, y + CAR_HEIGHT - 5, 10, 5);
        ctx.fillRect(x + 10, y + CAR_HEIGHT - 5, 10, 5);
        
        ctx.restore();
    }

    function update() {
        if (!gameActive) return;

        frame++;
        if (frame % 60 === 0) {
            let lane = Math.floor(Math.random() * 3);
            enemies.push({ x: LANES[lane], y: -100 });
            score++;
            if (score % 5 === 0) speed += 0.5;
        }

        enemies.forEach((enemy, index) => {
            enemy.y += speed;

            // Collision Check
            if (
                enemy.x === player.x &&
                enemy.y + CAR_HEIGHT > player.y &&
                enemy.y < player.y + CAR_HEIGHT
            ) {
                gameOver();
            }

            // Remove off-screen enemies
            if (enemy.y > 600) enemies.splice(index, 1);
        });
    }

    function draw() {
        // Background
        ctx.fillStyle = '#050505';
        ctx.fillRect(0, 0, canvas.width, canvas.height);

        // Lane Lines
        ctx.strokeStyle = '#222';
        ctx.setLineDash([20, 20]);
        ctx.beginPath();
        ctx.moveTo(100, 0); ctx.lineTo(100, 600);
        ctx.moveTo(225, 0); ctx.lineTo(225, 600);
        ctx.stroke();

        // UI Text
        ctx.setLineDash([]);
        ctx.font = 'bold 18px Courier New';
        ctx.fillStyle = '#0088aa';
        ctx.fillText(`SCORE: ${score}`, 20, 40);
        ctx.fillText(`BEST: ${bestScore}`, 220, 40);

        // Player
        drawCar(player.x, player.y, '#990099', true);

        // Enemies
        enemies.forEach(enemy => {
            drawCar(enemy.x, enemy.y, '#660066');
        });
    }

    function gameOver() {
        gameActive = false;
        if (score > bestScore) {
            bestScore = score;
            localStorage.setItem('nitroBest', bestScore);
        }
        overlay.style.display = 'flex';
        document.querySelector('h1').innerText = "CRASHED!";
    }

    function animate() {
        if (!gameActive) return;
        update();
        draw();
        requestAnimationFrame(animate);
    }

    // Controls
    window.addEventListener('keydown', (e) => {
        if (!gameActive) return;
        if (e.key === 'ArrowLeft' && player.lane > 0) player.lane--;
        if (e.key === 'ArrowRight' && player.lane < 2) player.lane++;
        player.x = LANES[player.lane];
    });

    // Touch support
    canvas.addEventListener('touchstart', (e) => {
        if (!gameActive) return;
        const touchX = e.touches[0].clientX - canvas.getBoundingClientRect().left;
        if (touchX < canvas.width / 2 && player.lane > 0) player.lane--;
        else if (touchX > canvas.width / 2 && player.lane < 2) player.lane++;
        player.x = LANES[player.lane];
    });

    startBtn.addEventListener('click', init);

    // Initial draw
    draw();
</script>
