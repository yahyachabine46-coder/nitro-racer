    <title>Nitro Racer: Smooth Cycle</title>
    <style>
        body { background: #000; display: flex; justify-content: center; align-items: center; height: 100vh; margin: 0; font-family: 'Arial', sans-serif; overflow: hidden; }
        #container { position: relative; width: 360px; height: 600px; border-radius: 30px; border: 4px solid #111; overflow: hidden; box-shadow: 0 0 50px rgba(0,0,0,1); }
        canvas { display: block; }
        #ui { position: absolute; top: 0; left: 0; width: 100%; height: 100%; display: flex; flex-direction: column; align-items: center; justify-content: center; background: rgba(0,0,0,0.8); z-index: 10; transition: opacity 0.5s; }
        h1 { color: #00f2ff; text-shadow: 0 0 20px #00f2ff; font-size: 32px; letter-spacing: 2px; }
        button { background: #ff00ff; color: white; border: none; padding: 18px 50px; border-radius: 50px; font-weight: bold; font-size: 20px; cursor: pointer; box-shadow: 0 0 25px #ff00ff; transition: transform 0.2s; }
        button:hover { transform: scale(1.1); }
        .score { position: absolute; top: 25px; left: 25px; color: #00f2ff; font-weight: bold; font-size: 20px; pointer-events: none; z-index: 5; text-shadow: 0 2px 4px #000; }
    </style>

<body>

<div id="container">
    <div class="score" id="sBox">SCORE: 0</div>
    <canvas id="gameCanvas" width="360" height="600"></canvas>
    <div id="ui">
        <h1 id="title">NITRO RACER</h1>
        <button onclick="play()">START DRIVE</button>
    </div>
</div>

<script>
    const canvas = document.getElementById('gameCanvas');
    const ctx = canvas.getContext('2d');
    const ui = document.getElementById('ui');
    const sBox = document.getElementById('sBox');

    // --- GAME SETTINGS ---
    const LANES = [75, 180, 285];
    let player = { targetLane: 1, x: 180, lerp: 0.15 };
    let enemies = [];
    let score = 0;
    let speed = 7;
    let active = false;
    let time = 0; // 0 to Math.PI * 2
    let loop;

    function drawCar(x, y, color, isPlayer) {
        const isNight = Math.sin(time) < -0.2;
        ctx.save();
        
        // Glow/Underglow
        ctx.shadowBlur = isNight ? 25 : 10;
        ctx.shadowColor = color;
        ctx.fillStyle = color;
        
        // Body
        ctx.beginPath();
        ctx.roundRect(x - 24, y, 48, 88, 12);
        ctx.fill();

        // Windshield
        ctx.shadowBlur = 0;
        ctx.fillStyle = "rgba(0,0,0,0.8)";
        ctx.fillRect(x - 18, y + 15, 36, 22);

        // Lights
        if (isNight) {
            // Headlights
            ctx.fillStyle = "rgba(255, 255, 220, 0.3)";
            ctx.beginPath();
            ctx.moveTo(x - 20, y); ctx.lineTo(x - 50, y - 80); ctx.lineTo(x + 50, y - 80); ctx.lineTo(x + 20, y);
            ctx.fill();
        }

        // Tail lights
        ctx.fillStyle = "#ff1111";
        ctx.fillRect(x - 18, y + 80, 8, 4);
        ctx.fillRect(x + 10, y + 80, 8, 4);
        
        ctx.restore();
    }

    function engine() {
        if (!active) return;
        time += 0.002; // Day/Night Cycle speed

        // 1. DYNAMIC ENVIRONMENT LERP
        const nightVal = (Math.sin(time) + 1) / 2; // 0 to 1
        // Background sky lerp (Blue to Black)
        const skyR = Math.floor(5 * nightVal);
        const skyG = Math.floor(15 * nightVal);
        const skyB = Math.floor(40 * nightVal);
        ctx.fillStyle = `rgb(${skyR}, ${skyG}, ${skyB})`;
        ctx.fillRect(0, 0, 360, 600);

        // Road lerp
        const roadGray = Math.floor(40 - (30 * nightVal));
        ctx.fillStyle = `rgb(${roadGray}, ${roadGray}, ${roadGray + 5})`;
        ctx.fillRect(25, 0, 310, 600);

        // Moving Lines
        ctx.strokeStyle = nightVal > 0.7 ? "#222" : "#444";
        ctx.setLineDash([40, 60]);
        ctx.lineDashOffset = -(Date.now() / 10) % 100;
        [120, 240].forEach(lx => {
            ctx.beginPath(); ctx.moveTo(lx, 0); ctx.lineTo(lx, 600); ctx.stroke();
        });

        // 2. SMOOTH PLAYER LERP
        const destX = LANES[player.targetLane];
        player.x += (destX - player.x) * player.lerp;

        // 3. ENEMIES
        if (Math.random() < 0.02) {
            enemies.push({ x: LANES[Math.floor(Math.random() * 3)], y: -100 });
        }

        for (let i = enemies.length - 1; i >= 0; i--) {
            let e = enemies[i];
            e.y += speed;
            drawCar(e.x, e.y, "#ff00ff", false);

            // Smooth Collision Detection
            if (Math.abs(e.x - player.x) < 40 && e.y + 80 > 480 && e.y < 560) {
                gameOver();
            }

            if (e.y > 650) {
                enemies.splice(i, 1);
                score++;
                sBox.innerText = "SCORE: " + score;
                speed += 0.05;
            }
        }

        drawCar(player.x, 480, "#00f2ff", true);
        loop = requestAnimationFrame(engine);
    }

    function play() {
        cancelAnimationFrame(loop);
        active = true;
        score = 0; speed = 7; enemies = []; time = 0;
        sBox.innerText = "SCORE: 0";
        ui.style.opacity = "0";
        setTimeout(() => ui.style.display = 'none', 500);
        engine();
    }

    function gameOver() {
        active = false;
        ui.style.display = 'flex';
        ui.style.opacity = "1";
        document.getElementById('title').innerText = "WASTED";
    }

    // --- CONTROLS ---
    window.addEventListener('keydown', e => {
        if (e.key === "ArrowLeft" && player.targetLane > 0) player.targetLane--;
        if (e.key === "ArrowRight" && player.targetLane < 2) player.targetLane++;
    });

    canvas.addEventListener('touchstart', e => {
        const touchX = e.touches[0].clientX - canvas.getBoundingClientRect().left;
        if (touchX < 180 && player.targetLane > 0) player.targetLane--;
        else if (touchX > 180 && player.targetLane < 2) player.targetLane++;
    });
</script>
